561 static inline bool looks_like_ip_address(const char *bucket)
562 {
563  int num_periods = 0;
564  bool expect_period = false;
565  for (const char *b = bucket; *b; ++b) {
566  if (*b == '.') {
567  if (!expect_period)
568  return false;
569  ++num_periods;
570  if (num_periods > 3)
571  return false;
572  expect_period = false;
573  }
574  else if (isdigit(*b)) {
575  expect_period = true;
576  }
577  else {
578  return false;
579  }
580  }
581  return (num_periods == 3);
582 }

594 static inline int valid_s3_bucket_name(const string& name, bool relaxed=false)
595 {
596  // This function enforces Amazon's spec for bucket names.
597  // (The requirements, not the recommendations.)
598  int len = name.size();
599  if (len < 3) {
600  // Name too short
601  return -ERR_INVALID_BUCKET_NAME;
602  } else if (len > 255) {
603  // Name too long
604  return -ERR_INVALID_BUCKET_NAME;
605  }
606 
607  // bucket names must start with a number, letter, or underscore
608  if (!(isalpha(name[0]) || isdigit(name[0]))) {
609  if (!relaxed)
610  return -ERR_INVALID_BUCKET_NAME;
611  else if (!(name[0] == '_' || name[0] == '.' || name[0] == '-'))
612  return -ERR_INVALID_BUCKET_NAME;
613  }
614 
615  for (const char *s = name.c_str(); *s; ++s) {
616  char c = *s;
617  if (isdigit(c) || (c == '.'))
618  continue;
619  if (isalpha(c))
620  continue;
621  if ((c == '-') || (c == '_'))
622  continue;
623  // Invalid character
624  return -ERR_INVALID_BUCKET_NAME;
625  }
626 
627  if (looks_like_ip_address(name.c_str()))
628  return -ERR_INVALID_BUCKET_NAME;
629 
630  return 0;
631 }
