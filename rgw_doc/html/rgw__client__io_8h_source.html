91  std::streambuf::int_type underflow() {
92  if (gptr() < egptr()) {
93  return traits_type::to_int_type(*gptr());
94  }
95 
96  char * const base = buffer.data();
97  char * start;
98 
99  if (nullptr != eback()) {
100  /* We need to skip moving bytes on first underflow. In such case
101  * there is simply no previous data we should preserve for unget()
102  * or something similar. */
103  std::memmove(base, egptr() - putback_size, putback_size);
104  start = base + putback_size;
105  } else {
106  start = base;
107  }
108 
109  int read_len;
110  int ret = sio.read(base, window_size, &read_len);
111  if (ret < 0 || 0 == read_len) {
112  return traits_type::eof();
113  }
114 
115  setg(base, start, start + read_len);
116 
117  return traits_type::to_int_type(*gptr());
118  }
