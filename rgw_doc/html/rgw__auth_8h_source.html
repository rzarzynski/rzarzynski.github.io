222 /* Abstract class for authentication backends (auth engines) in RadosGW.
223  *
224  * An engine is supposed only to:
225  * - authenticate (not authorize!) a given request basing on req_state,
226  * - provide an upper layer with RGWAuthApplier to commit all changes to
227  * data structures (like req_state) and to the RADOS store (creating
228  * an account, synchronizing user personal info).
229  * Auth engine MUST NOT make any changes to req_state nor RADOS store.
230  *
231  * Separation between authentication and global state modification has been
232  * introduced because many auth engines are perfectly orthogonal to applier
233  * and thus they can be decoupled. Additional motivation is clearly distinguish
234  * all places which can modify underlying data structures. */
235 class RGWAuthEngine {
236 protected:
237  CephContext * const cct;
238 
239  RGWAuthEngine(CephContext * const cct)
240  : cct(cct) {
241  }
242  /* Make the engines non-copyable and non-moveable due to const-correctness
243  * and aggregating applier factories less costly and error-prone. */
244  RGWAuthEngine(const RGWAuthEngine&) = delete;
245  RGWAuthEngine& operator=(const RGWAuthEngine&) = delete;
246 
247 public:
248  /* Get name of the auth engine. */
249  virtual const char* get_name() const noexcept = 0;
250 
251  /* Fast, non-throwing method for screening whether a concrete engine may
252  * be interested in handling a specific request. */
253  virtual bool is_applicable() const noexcept = 0;
254 
255  /* Throwing method for identity verification. When the check is positive
256  * an implementation should return RGWAuthApplier::aplptr_t containing
257  * a non-null pointer to object conforming the RGWAuthApplier interface.
258  * Otherwise, the authentication is treated as failed.
259  * An error may be signalised by throwing an exception of int type with
260  * errno value inside. Those value are always negative. */
261  virtual RGWAuthApplier::aplptr_t authenticate() const = 0;
262 
263  virtual ~RGWAuthEngine() {};
264 };
