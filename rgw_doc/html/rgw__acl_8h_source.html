217  bool is_match(std::string http_referer) const {
218  if (http_referer == url_spec) {
219  return true;
220  }
221 
222  if (http_referer.length() < url_spec.length()) {
223  return false;
224  }
225 
226  if ('.' == url_spec[0]) {
227  /* Wildcard support: a referer matches the spec when its last char are
228  * perfectly equal to spec. */
229  return !http_referer.compare(http_referer.length() - url_spec.length(),
230  url_spec.length(), url_spec);
231  }
232 
233  return false;
234  }
