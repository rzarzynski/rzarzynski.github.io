129 template <typename T>
130 void RGWThirdPartyAccountAuthApplier<T>::load_acct_info(RGWUserInfo& user_info) const
131 {
132  if (UNKNOWN_ACCT == acct_user_override) {
133  /* There is no override specified by the upper layer. This means that we'll
134  * load the account owned by the authenticated identity (aka auth_user). */
135  RGWDecoratingAuthApplier<T>::load_acct_info(user_info);
136  } else if (RGWDecoratingAuthApplier<T>::is_owner_of(acct_user_override)) {
137  /* The override has been specified but the account belongs to the authenticated
138  * identity. We may safely forward the call to a next stage. */
139  RGWDecoratingAuthApplier<T>::load_acct_info(user_info);
140  } else {
141  /* Compatibility mechanism for multi-tenancy. For more details refer to
142  * load_acct_info method of RGWRemoteAuthApplier. */
143  if (acct_user_override.tenant.empty()) {
144  const rgw_user tenanted_uid(acct_user_override.id, acct_user_override.id);
145 
146  if (rgw_get_user_info_by_uid(store, tenanted_uid, user_info) >= 0) {
147  /* Succeeded. */
148  return;
149  }
150  }
151 
152  int ret = rgw_get_user_info_by_uid(store, acct_user_override, user_info);
153  if (ret < 0) {
154  /* We aren't trying to recover from ENOENT here. It's supposed that creating
155  * someone else's account isn't a thing we want to support in this filter. */
156  throw ret;
157  }
158 
159  }
160 }
